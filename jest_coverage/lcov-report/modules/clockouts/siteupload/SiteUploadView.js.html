import React, { useState, useEffect } from 'react';
import {ImageBackground, View, ActivityIndicator, Alert} from 'react-native';
import R from 'ramda';
import {Container, Button, Left, Right, Text, Body} from 'native-base';
import SMHeader from "../../../components/SMHeader";
import SMContent from "../../../components/SMContent";
import SMFooter from "../../../components/SMFooter";
import SUPicture from "./PictureTypeSiteUpload";
 
const SiteUploadView = (props) => {
  const [pendingTemplate] = useState(props.pendingTemplate);
  const [promptGroup] = useState(props.template.PromptGroup);
  const [isAllLeft, setIsAllLeft] = useState(true);
  const [isEdit, setIsEdit] = useState(false);
  const [leftCheckObject, setLeftCheckObject] = useState(props.template.SitephotosTemplatePrompt);
  const [siteUploadPhotoTemplate, setSiteUploadPhotoTemplate] = useState([]);
 
  useEffect(() => {
    if (props.navigation.getParam('isPending')) {
      setSiteUploadPhotoTemplate(props.template.SitephotosTemplatePrompt);
    } else {
      if (pendingTemplate[props.template.id]) {
        Alert.alert(
          'Saved session exist.',
          'Do you want to continue a saved session?',
          [
            {
              text: 'Cancel', style: 'cancel',
              onPress: () => {
                setSiteUploadPhotoTemplate(props.template.SitephotosTemplatePrompt);
              },
            },
            {
              text: 'OK',
              onPress: () => {
                setSiteUploadPhotoTemplate(pendingTemplate[props.template.id].SitephotosTemplatePrompt);
              }
            },
          ],
          {cancelable: false},
        );
      } else {
        setSiteUploadPhotoTemplate(props.template.SitephotosTemplatePrompt);
      }
    }
  }, [pendingTemplate]);
  // Get promptGroup from id
  const getPromptGroup = (id) => {
    let selectedPromptGroup = null;
    if (!R.isEmpty(promptGroup)) {
      selectedPromptGroup = R.find(R.propEq('id', id))(promptGroup);
    }
    return selectedPromptGroup
  };
 
  const submitData = () => {
    const currentTime = new Date().getTime();
    let requestBody = props.template;
 
    // Add prompt array in PromptGroup for request body
    let _promptGroup = [];
    Object.keys(promptGroup).forEach(key => {
      let item = promptGroup[key];
      item.undoneprompts = 0;
      item.undone = false;
      item.prompts = R.filter(R.propEq('prompt_group_id', item.id), siteUploadPhotoTemplate);
      _promptGroup.push({...item});
    });
    requestBody.PromptGroup = _promptGroup;
    requestBody.SitephotosTemplatePrompt = siteUploadPhotoTemplate;
    // delete requestBody.SitephotosTemplatePrompt; // Remove it in request body because does not need to send it to server
 
    // Add photos in request body
    let _photos = [];
    Object.keys(siteUploadPhotoTemplate).forEach(key => {
      let item = siteUploadPhotoTemplate[key];
      if (!R.isNil(item.photos)) _photos = R.concat(_photos, item.photos);
    });
    if (!R.isEmpty(_photos)) requestBody.photos = _photos;
 
    requestBody.workOrderNumber = props.project['WO#'];
    requestBody.created = currentTime;
    requestBody.uploaded = false;
    requestBody.completed = false;
    requestBody.uploading = true;
    setIsEdit(false);
    return requestBody;
  };
 
  // Save data into local without submitting to server.
  const onSaveToLocal = () => {
    props.saveDataLocal(submitData());
    setIsEdit(false);
  };
  // Save data to the server(Save action).
  const onSaveData = async () => {
    await props.saveData(submitData());
    props.navigation.navigate('CloseOuts');
  };
 
  // Finish data to the server(Finish action).
  const onFinishData = async () => {
    await props.finishData(submitData());
    props.navigation.navigate('CloseOuts');
  };
 
  const setPhotoData = async (photoData, templateId) => {
    const tIndex = R.findIndex(R.propEq('id', templateId))(siteUploadPhotoTemplate);
    siteUploadPhotoTemplate[tIndex].hasValue = !R.isEmpty(photoData);
    siteUploadPhotoTemplate[tIndex].isFreeText = false;
    siteUploadPhotoTemplate[tIndex].isSelect = false;
    siteUploadPhotoTemplate[tIndex].isNumber = false;
    siteUploadPhotoTemplate[tIndex].isPicture = true;
    siteUploadPhotoTemplate[tIndex].photos = photoData;
    setSiteUploadPhotoTemplate(siteUploadPhotoTemplate);
    setIsEdit(true);
    await props.saveData(submitData());
  };
 
  const setNASelect = async (value, templateId) => {
    const tIndex = R.findIndex(R.propEq('id', templateId))(siteUploadPhotoTemplate);
    siteUploadPhotoTemplate[tIndex].isNA = value;
    siteUploadPhotoTemplate[tIndex].na = value;
    setSiteUploadPhotoTemplate(siteUploadPhotoTemplate);
    setIsEdit(true);
    await props.saveData(submitData());
  };
 
  const updateIsLeft = (value, templateId) => {
    const tIndex = R.findIndex(R.propEq('id', templateId))(siteUploadPhotoTemplate);
    // TODO: consider of a field for checking left
    leftCheckObject[tIndex]["isLeft"] = value;
    setLeftCheckObject(leftCheckObject);
    setIsAllLeft(R.findLastIndex(R.propEq('isLeft', true))(leftCheckObject) !== -1);
  };
 
  const goPrevious = () => {
    if (isEdit) {
      Alert.alert(
        'Some data has been changed',
        'Do you want to save this in Local Storage?',
        [
          {
            text: 'Cancel', style: 'cancel',
            onPress: () => {},
          },
          {
            text: 'Exit without saving', style: 'cancel',
            onPress: () => {
              props.navigation.pop();
            },
          },
          {
            text: 'Exit and save',
            onPress: () => {
              onSaveData();
              props.navigation.pop();
            }
          },
        ],
        {cancelable: false},
      );
    } else {
      props.navigation.pop();
    }
  };
 
  return (
    <Container>
      <SMHeader
        title={props.template.name}
        navigation={props.navigation}
        isMenu={false}
        isRefresh={false}
        isHome isBack
        onGoBack={() => goPrevious()}
      />
      <SMContent>
        {!R.isEmpty(siteUploadPhotoTemplate) &&
          siteUploadPhotoTemplate.map((item, index) => {
            switch(item.type) {
              case "Picture":
                return (
                  <SUPicture
                    key={index}
                    promptGroup={getPromptGroup(item['prompt_group_id'])}
                    siteUploadTemplate={item}
                    onSetPhoto = {(data, t_id) => setPhotoData(data, t_id)}
                    onSetNASelect = {(value, t_id) => setNASelect(value, t_id)}
                    onUpdateIsLeft = {(value, t_id) => updateIsLeft(value, t_id)}
                  />
                );
              default: return "";
            }
          })
        }
      </SMContent>
      <SMFooter>
        <Left>
          {(!props.loadingStatus) && <Button transparent light onPress={() => goPrevious()}>
            <Text>Cancel</Text>
          </Button>}
        </Left>
        <Body>
          <Button style={{flex: 1, alignItems: 'center', justifyContent: 'center',}} transparent light onPress={() => onSaveData()} disabled={props.loadingStatus}>
            {(props.loadingStatus) && <ActivityIndicator />}
            {(!props.loadingStatus) && <Text style={{flex: 1, textAlign: 'center'}}>Save</Text>}
          </Button>
        </Body>
        <Right>
          {(!props.loadingStatus) && <Button transparent light disabled={isAllLeft} onPress={() => onFinishData()}>
            <Text>Complete</Text>
          </Button>}
        </Right>
      </SMFooter>
    </Container>
  );
};
 
 
export default SiteUploadView;
