const sendEmailData = (data) => async dispatch => {
  const sendAction = {
    do: () => createAction(CLOSEOUT_EMAILSEND.DO, {}),
    pending: response => createAction(CLOSEOUT_EMAILSEND.PENDING, {data: response}),
    failed: (error, data) => createAction(CLOSEOUT_EMAILSEND.FAILED, {error, data}),
  };
  try {
    dispatch(sendAction.do());
    let response = await post(global.apiEndpoints.sitephotosupload, data);
    data.handle = response.handle;
    let photos = (data.photos && !R.isEmpty(data.photos))
      ? R.filter((item) => !item.uploadStatus || item.uploadStatus !== 'success', data.photos) : [];
    if (photos.length) {
      let uploadQue = [];
      photos.map((item, index) => {
        item.handle = response.handle;
        uploadQue.push(upload(global.apiEndpoints.sitephotosimgupload, item));
        photos[index].uploadStatus = "failed";
      });
 
      let uploadResponse = await Promise.all(uploadQue);
 
      if (!R.isEmpty(uploadResponse)) {
        uploadResponse.map((item, index) => {
          photos[index].uploadStatus = "success";
        })
      }
      data.photos = photos;
    }
    dispatch(sendAction.pending(data));
  } catch (e) {
    console.log(e);
    dispatch(sendAction.failed(transformNetworkError(e), data));
  }
};

const finishData = (data) => async dispatch => {
  const sendAction = {
    do: () => createAction(CLOSEOUT_SAVEDATA.DO, {}),
    pending: data => createAction(CLOSEOUT_SAVEDATA.PENDING, {data}),
    success: data => createAction(CLOSEOUT_SAVEDATA.SUCCESS, {data}),
    failed: (error, data) => createAction(CLOSEOUT_SAVEDATA.FAILED, {error, data}),
  };
  try {
    dispatch(sendAction.do());
    const response1 = await post(global.apiEndpoints.sitephotosupload, data);
    dispatch(sendAction.pending(data));
    if (response1.handle)
      await get(`${global.apiEndpoints.sitephotosfinish}/${response1.handle}`);
    dispatch(sendAction.success(data));
  } catch (e) {
    console.log(e);
    dispatch(sendAction.failed(transformNetworkError(e), data));
  }
};
