import {
  createTypes,
  createAction,
  transformNetworkError,
  createSetTypes,
} from '../../utils/actions';
 
import {get, getWithNoHeader, getDigestData} from '../../utils/request';
 
const DIGEST = createTypes('SMCOSMINC_DIGEST');
const SYNC = createTypes('SMCOSMINC_SYNC');
const TEMPLATE_UPDATE = createSetTypes('TEMPLATE_UPDATE');
 
const getDigest = () => async dispatch => {
    const digestAction = {
        do: () => createAction(DIGEST.DO, {}),
        success: response => createAction(DIGEST.SUCCESS, {urls: response}),
        failed: () => createAction(DIGEST.FAILED, {}),
    };
    try {
        dispatch(digestAction.do());
        const response = await getWithNoHeader('/apis/digest');
        dispatch(digestAction.success(response));
    } catch (e) {
        dispatch(digestAction.failed(transformNetworkError(e)));
    }
};
 
const getSyncData = () => async dispatch => {
  const syncAction = {
    do: () => createAction(SYNC.DO, {}),
    success: response => createAction(SYNC.SUCCESS, {data: response}),
    failed: () => createAction(SYNC.FAILED, {}),
  };
  try {
    dispatch(syncAction.do());
    const response = await get(global.apiEndpoints.offlineData + '/1/0/1/1/1/0/0');
    dispatch(syncAction.success(response));
  } catch (e) {
    console.log(e);
    dispatch(syncAction.failed(transformNetworkError(e)));
  }
};
 
const updateTemplateData = (id, data) => dispatch => {
  const setEmailAction = {
    setData: (id, data) => createAction(TEMPLATE_UPDATE.SET, {data, id}),
    clearData: () => createAction(TEMPLATE_UPDATE.CLEAR, null)
  };
  try {
    dispatch(setEmailAction.setData(id, data));
  } catch (e) {
    dispatch(setEmailAction.clearData());
  }
};
 
 
export {DIGEST, SYNC, TEMPLATE_UPDATE, getSyncData, getDigest, updateTemplateData};
