import {
  createTypes,
  createAction,
  transformNetworkError,
} from '../../utils/actions';
 
import {getWithHeader, setHeader} from '../../utils/request';
 
const LOGIN = createTypes('SMCOSMINC_LOGIN');
const LOGOUT = createTypes('SMCOSMINC_LOGOUT');
 
const logIn = formData => {
  return async dispatch => {
    const loginAction = {
      do: () => createAction(LOGIN.DO, {}),
      success: (response, authData) => createAction(LOGIN.SUCCESS, {userData: response, authData: authData}),
      failed: (error) => createAction(LOGIN.FAILED, error),
    };
    try {
      dispatch(loginAction.do());
      const authData = `username=${formData.email}&password=${formData.password}`;
      const resp = await getWithHeader('/apis/credcheck', authData);
      // await LocalStorage.set(authHeaderStorageKey, `username=${formData.email}&password=${formData.password}`);
      // setHeader(authData);
      global.Authorization = authData;
      dispatch(loginAction.success(resp, authData));
    } catch (e) {
      console.log('transformNetworkError', e);
      dispatch(loginAction.failed(transformNetworkError(e)));
    }
  };
};
 
 
const logOut = () => {
  return async dispatch => {
    const logoutAction = {
      do: () => createAction(LOGOUT.DO, {}),
      success: () => createAction(LOGOUT.SUCCESS, {}),
      failed: () => createAction(LOGOUT.FAILED, {}),
    };
    try {
      dispatch(logoutAction.do());
      dispatch(logoutAction.success());
      // await AsyncStorage.clear();
    } catch (e) {
      dispatch(logoutAction.failed(e));
    }
 
  };
};
 
 
export {LOGIN, LOGOUT, logIn, logOut};
